  <script>
    const socket = io();
    let userId;

    document.getElementById('registerForm').addEventListener('submit', async (e) => {
      e.preventDefault();
      const name = document.getElementById('name').value;
      const email = document.getElementById('email').value;
      const res = await fetch('/register', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name, email })
      });
      const data = await res.json();
      userId = data.id;
      document.getElementById('registerForm').style.display = 'none';
      document.getElementById('dashboard').style.display = 'block';
      loadDashboard();
    });

    async function loadDashboard() {
      const res = await fetch(`/dashboard/${userId}`);
      const loads = await res.json();
      const loadsDiv = document.getElementById('loads');
      loadsDiv.innerHTML = '';
      loads.forEach(load => {
        const div = document.createElement('div');
        div.className = 'load';
        div.innerHTML = `
          <p>Load ID: ${load.id}</p>
          <p>From: \( {load.origin} to \){load.destination}</p>
          <p>Cargo: ${load.cargo}</p>
          <p>Status: ${load.status}</p>
          <a href="/paperwork/${load.id}" download>Download BOL</a>
        `;
        loadsDiv.appendChild(div);
      });
    }

    socket.on('update', (data) => {
      if (data.driverId === userId) loadDashboard();
    });

    function speakStatus() {
      const msg = new SpeechSynthesisUtterance('Your loads are being updated.');
      window.speechSynthesis.speak(msg);
    }

    // Voice input (browser support)
    if ('webkitSpeechRecognition' in window) {
      const recognition = new webkitSpeechRecognition();
      recognition.onresult = (e) => {
        document.getElementById('name').value = e.results[0][0].transcript;
      };
      // Add button to trigger if needed
    }
  </script>
